Install

Install with npm:

$ npm install --save extend-shallow

Usage

var extend = require('extend-shallow');
extend({a: 'b'}, {c: 'd'})
//=> {a: 'b', c: 'd'}

Pass an empty object to shallow clone:

var obj = {};
extend(obj, {a: 'b'}, {c: 'd'})
//=> {a: 'b', c: 'd'}
