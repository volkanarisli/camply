\subsection*{Install}


\begin{DoxyCode}

\end{DoxyCode}


\subsection*{Features}


\begin{DoxyItemize}
\item E\+S5 compatible
\item works in node.\+js (0.\+10+) and browsers (I\+E9+)
\item checks equality of Date and Reg\+Exp objects by value.
\end{DoxyItemize}

\subsection*{Usage}


\begin{DoxyCode}
var equal = require('fast-deep-equal');
console.log(equal(\{foo: 'bar'\}, \{foo: 'bar'\})); // true
\end{DoxyCode}
