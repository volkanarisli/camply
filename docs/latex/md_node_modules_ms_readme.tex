Use this package to easily convert various time formats to milliseconds.

\subsection*{Examples}


\begin{DoxyCode}
ms('2 days')  // 172800000
ms('1d')      // 86400000
ms('10h')     // 36000000
ms('2.5 hrs') // 9000000
ms('2h')      // 7200000
ms('1m')      // 60000
ms('5s')      // 5000
ms('1y')      // 31557600000
ms('100')     // 100
\end{DoxyCode}


\subsubsection*{Convert from milliseconds}


\begin{DoxyCode}
ms(60000)             // "1m"
ms(2 * 60000)         // "2m"
ms(ms('10 hours'))    // "10h"
\end{DoxyCode}


\subsubsection*{Time format written-\/out}


\begin{DoxyCode}
ms(60000, \{ long: true \})             // "1 minute"
ms(2 * 60000, \{ long: true \})         // "2 minutes"
ms(ms('10 hours'), \{ long: true \})    // "10 hours"
\end{DoxyCode}


\subsection*{Features}


\begin{DoxyItemize}
\item Works both in \href{https://nodejs.org}{\tt node} and in the browser.
\item If a number is supplied to {\ttfamily ms}, a string with a unit is returned.
\item If a string that contains the number is supplied, it returns it as a number (e.\+g.\+: it returns {\ttfamily 100} for `\textquotesingle{}100'\`{}).
\item If you pass a string with a number and a valid unit, the number of equivalent ms is returned.
\end{DoxyItemize}
