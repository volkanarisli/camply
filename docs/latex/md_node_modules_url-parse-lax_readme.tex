var urlParseLax = require('url-parse-lax');

urlParseLax('sindresorhus.com');
/*
\{
  protocol: null,
  slashes: true,
  auth: null,
  host: 'sindresorhus.com',
  port: null,
  hostname: 'sindresorhus.com',
  hash: null,
  search: null,
  query: null,
  pathname: '/',
  path: '/',
  href: 'http://sindresorhus.com/'
\}
*/

\{
  protocol: null,
  slashes: true,
  auth: null,
  host: '[2001:db8::]:8000',
  port: '8000',
  hostname: '2001:db8::',

  search: null,
  query: null,
  pathname: '/',
  path: '/',
  href: 'http://[2001:db8::]:8000/'
